using System;

class Program
{
    static void Main()
    {
        Console.Write("Введите возраст: ");
        int age = int.Parse(Console.ReadLine());

        // 2. if-else if-else для проверки корректности возраста
        string category = "";
        if (age < 0)
        {
            category = "Некорректный возраст";
        }
        else if (age <= 1)
        {
            category = "Младенец";
        }
        else
        {
            // 1. switch для классификации категорий
            switch (age)
            {
                case int n when (n >= 2 && n <= 12):
                    category = "Ребенок";
                    break;
                case int n when (n >= 13 && n <= 19):
                    category = "Подросток";
                    break;
                case int n when (n >= 20 && n <= 64):
                    category = "Взрослый";
                    break;
                default:
                    category = "Пожилой человек";
                    break;
            }
        }

        // 3. Тернарный оператор для формирования финальной строки
        string result = category == "Некорректный возраст" ? 
                       "Некорректный возраст" : 
                       $"Категория: {category}";

        // 4. Вывод ровно один раз
        Console.WriteLine(result);
    }
}

Соответствие требованиям
if-else if-else: проверка age < 0 и age <= 1
​

switch с when: все категории 2-12, 13-19, 20-64, 65+
​

Тернарный оператор: формирование Категория: X или ошибка
​

Один Console.WriteLine: финальный вывод
​
