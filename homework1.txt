using System;

class Program
{
    static void Main()
    {
        // 1. Объявление переменных для оценок с явным типом int
        int mathGrade = 4;
        int physicsGrade = 3;
        int csGrade = 5;

        // 3. var для суммы оценок (неявная типизация)
        var totalGrade = mathGrade + physicsGrade + csGrade;

        // 4. Вычисление среднего балла как double
        double averageGrade = (double)totalGrade / 3;

        // 5. Логическая переменная для проверки сдачи сессии
        bool sessionPassed = averageGrade >= 3.0;

        // 6. Формирование строки отчёта
        string report = $"Оценки: Математика={mathGrade}, Физика={physicsGrade}, Информатика={csGrade}, " +
                       $"Средний балл={averageGrade:F2}, " +
                       $"Сессия {(sessionPassed ? "сдана" : "не сдана")}";

        // 7. Вывод результата
        Console.WriteLine(report);
    }
}
Результат выполнения
text
Оценки: Математика=4, Физика=3, Информатика=5, Средний балл=4.00, Сессия сдана
Соответствие требованиям
Явные типы int: mathGrade, physicsGrade, csGrade
​

var для суммы: totalGrade (неявно int)
​

Средний балл double: приведение (double)totalGrade / 3
​

Логика bool: averageGrade >= 3.0
​

Форматирование: {averageGrade:F2} для 2 знаков после запятой
​

Тернарный оператор: sessionPassed ? "сдана" : "не сдана"
​

Тестирование граничных случаев
text
math=2, physics=2, cs=2 → Средний=2.00, Сессия не сдана
math=5, physics=5, cs=5 → Средний=5.00, Сессия сдана  
math=3, physics=3, cs=3 → Средний=3.00, Сессия сдана
